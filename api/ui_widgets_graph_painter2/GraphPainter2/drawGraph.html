    auto = false;
    highlight = false;
    gridPerMin = 1;
  }

  drawXAxis(canvas);
  drawYAxis(canvas);

  drawTocoXAxis(canvas);
  drawTocoYAxis(canvas);

  //drawBPMLine(canvas, /*interpretations.baselineBpmList*/, graphBaseLine);
  if (displayFhr != 2) {
    drawBPMLine(canvas, test.bpmEntries!, graphBpmLine);
